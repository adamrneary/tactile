chart = new Tactile.Chart(padding: {top: 0, right: 0, bottom: 5, left: 0}, unstack: false)
  .element($("#example_view")[0])
  .data(data)
  .axes({x:{dimension: "linear"}})
  .setXFrame(frameVal)
  .addSeries [
    name: "enemies"
    renderer: "area"
    dotSize: 1
    sigfigs: 0
    isEditable: true
    afterDrag: (d, y, i, draggedSeries, graph) ->
      graph.data()[i].y = y
    color: "#c05020"
    tooltip: (d) ->
      d.y + " friends"
    dataTransform: (d) ->
      x: d.x
      y: d.y
  ,
    name: "friends"
    renderer: "area"
    sigfigs: 1
    color: "#6060c0"
    isEditable: true
    afterDrag: (d, y, i, draggedSeries, graph) ->
      graph.data()[i].z = y
    tooltip: (d) ->
      d.y + " friends"
    dataTransform: (d) ->
      x: d.x
      y: d.z
  ]

chart.render()
