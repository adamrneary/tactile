# create instance of chart
chart = new Tactile.Chart(unstack: true)
  .element($("#example_view")[0])


# Stack / Unstack behavior
stack.click (e) ->
  e.stopPropagation()
  stackGroup.find("button").removeClass("active")
  $(this).addClass("active")

  if currentChart in ['column', 'area']
    chart.stackTransition()

unstack.click (e) ->
  e.stopPropagation()
  stackGroup.find("button").removeClass("active")
  $(this).addClass("active")

  if currentChart in ['column', 'area']
    chart.unstackTransition()

# baserenderer for ''
baseRenderer = () ->
  chart.data(chartData)
    .addSeries([
      name: "reach actual"
      renderer: currentChart
      sigfigs: 0
      round: false
      color: "#c05020"
      fromBaseline: true
      tooltip: (d) ->
        d.y + " customers"

      dataTransform: (d) ->
        x: d.x
        y: d.z
    ,
      name: "planned"
      renderer: currentChart
      round: false
      color: "#6060c0"
      tooltip: (d) ->
        d.y + " planned"

      dataTransform: (d) ->
        x: d.x
        y: d.y
    ], {overwrite: true})
    .setXFrame(frameVal)
  axes = _.keys(chart.initAxes())
  unless ("x" in axes) and ("y" in axes)
    chart.axes
      x:
        dimension: 'time'
        frame: frameVal
      y:
        dimension: "linear"

  chart.render()


toColumn.click (e) ->
  currentChart = "column"
  e.stopPropagation()
  buttonGroup.find("button").removeClass("active")
  $(this).addClass("active")
  stackGroup.find("button").removeAttr("disabled")
  stack.removeClass("active")
  unstack.addClass("active")

  baseRenderer()

toLeaderboard.click (e) ->
  currentChart = "leaderboard"
  e.stopPropagation()
  buttonGroup.find("button").removeClass("active")
  $(this).addClass("active")
  stackGroup.find("button").attr("disabled", "disabled")

  generateData = (count) =>
    i = 0
    while i < count
      oldValue = leaderboardData[i].value
      leaderboardData[i].value = Math.floor Math.random() * 1000
      leaderboardData[i].change = leaderboardData[i].value - oldValue
      leaderboardData[i].barPosition = leaderboardData[i].value / 1000
      i++

    i = 0
    while i < count
      j = 0
      index = 0
      while j < count
        index++ if leaderboardData[j].value > leaderboardData[i].value
        j++
      leaderboardData[i].index = index
      i++

    leaderboardData

  chart.data(generateData(4))
    .addSeries([
      name: "enemies"
      renderer: "leaderboard"
      changeFormat: d3.format("f")
      valueFormat: d3.format("f")
      dataTransform: (d) ->
        label: d.label
        value: d.value
        change: d.change
        barPosition: d.barPosition
        index: d.index
    ], overwrite: true)
    .axes({})

  chart.render()

toPareto.click (e) ->
  currentChart = "pareto"
  e.stopPropagation()
  buttonGroup.find("button").removeClass("active")
  $(this).addClass("active")
  stackGroup.find("button").attr("disabled", "disabled")

  chart.data(paretoData)
    .addSeries([
        name: "Revenue"
        color: "#702AFF"
        renderer: 'column'
        wide: true
        round: true
        dataTransform: (d) ->
          x: d.x
          y: d.value
          name: d.name
        tooltip: (d) ->
          "#{d.name}: #{d.y}"
      ,
        name: 'Cumulative percentage'
        color: "#C20055"
        renderer: 'line'
        yAxis: 'y1'
        dataTransform: (d) ->
          x: d.x
          y: d.cumulativePercent
          name: d.name
        tooltip: (d) ->
          "#{d.name}: #{(d.y * 100).toFixed(2)}% (cumulative)"
      ], overwrite: true)
    .setAutoScale(true)
    .setMinXFrame(1)
    .axes
      y: {dimension: "linear"}
      y1: {dimension: "linear"}

  chart.render()
