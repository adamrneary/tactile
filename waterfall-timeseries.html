chart = new Tactile.Chart(min: 'auto')
  .element($("#example_view")[0])
  .data(data)
  .axes({x: {dimension: 'time'}, y: {dimension: "linear"}})
  .setXFrame(frameVal)
  .addSeries [
    name: "Customers"
    renderer: "waterfall"
    color: "#30878F"
    fromBaseline: true
    tooltip: (d) ->
      d.y + " customers"
    dataTransform: (d) ->
      x: d.period
      y: d.customers
  ,
    name: "New customers"
    renderer: "waterfall"
    color: "#8ac16f"
    tooltip: (d) ->
      d.y + " new customers"
    dataTransform: (d) ->
      x: d.period
      y: d.newCustomers
  ,
    name: "Churned customers"
    renderer: "waterfall"
    color: "#dc6e59"
    tooltip: (d) ->
      d.y + " churned customers"
    dataTransform: (d) ->
      x: d.period
      y: d.churnedCustomers
  ]

chart.render()
